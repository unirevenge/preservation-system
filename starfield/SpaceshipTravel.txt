; SpaceshipTravel.txt - Enhanced ship travel commands
cgf "Debug.Notification" "Spaceship Systems Online"

; Fuel and ship systems
player.setav SpaceshipFuel 100
player.setav ShieldMaxHealth 500
player.setav shieldregenrate 3

; Travel enhancements
SetGS fSpaceshipLootingDistanceDefault 1500
SetGS fSpaceshipMaxDockingDistance 1500
Setgs fMaxShipTransferDistance 1500

; Crew management
setav spaceshippassengerslots 20
setav SpaceshipCrewRating 50
setav SpaceshipCrewSlots 15

; Ship weapons and combat
player.setav SpaceshipMissileDamageMult 1.5
player.setav SpaceshipLaserDamageMult 1.5
player.setav SpaceshipBallisticDamageMult 1.5
player.setav SpaceshipParticleDamageMult 1.5

; Grav drive and travel
player.setav SpaceshipGravJumpFuelUseMult 0.5
SetGS fGravJumpTravelTimeMult 0.5

cgf "Debug.Notification" "Ship Travel Enhanced!"
